\ExplSyntaxOn

\tl_new:N \g_project_github_hash_tl

\cs_if_exist:NF \githubhash
  { \NewDocumentCommand{\githubhash}{}{} }

\cs_new_protected:Npn \fayorg_project_set_github_hash:n #1
  {
    \tl_gset:Nn \g_project_github_hash_tl { #1 }
    \RenewDocumentCommand{\githubhash}{}{ #1 }
  }

\cs_new_protected:Npn \fayorg_project_version:n #1
  {
    \tl_if_empty:NTF \g_project_github_hash_tl
      {
        \tl_if_blank:nF { #1 }
          { Version:~\texttt { #1 } \\ }
      }
      {
        Version:~\texttt { \tl_range:Nnn \g_project_github_hash_tl { 1 } { 7 } } \\
      }
  }

\keys_define:nn { project } {
    github_hash .code:n = { \fayorg_project_set_github_hash:n { #1 } },
    github hash .code:n = { \fayorg_project_set_github_hash:n { #1 } },
    githubhash .code:n = { \fayorg_project_set_github_hash:n { #1 } },
}

\NewDocumentCommand{\ProjectGitHubHash}{}
  { \tl_use:N \g_project_github_hash_tl }

\NewDocumentCommand{\ProjectVersion}{O{}}
  { \fayorg_project_version:n { #1 } }

\ExplSyntaxOff
